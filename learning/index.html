💻 コードスニペット

Python

Union-Find (素集合データ構造)

class UnionFind: def __init__(self, n): self.parent = list(range(n)) self.rank = [0] * n self.size = [1] * n def find(self, x): if self.parent[x] != x: self.parent[x] = self.find(self.parent[x]) return self.parent[x] def union(self, x, y): rx, ry = self.find(x), self.find(y) if rx == ry: return False if self.rank[rx] < self.rank[ry]: rx, ry = ry, rx self.parent[ry] = rx self.size[rx] += self.size[ry] if self.rank[rx] == self.rank[ry]: self.rank[rx] += 1 return True
JavaScript

debounce 関数

function debounce(fn, delay = 300) { let timer; return (...args) => { clearTimeout(timer); timer = setTimeout(() => fn(...args), delay); }; } // 使用例 const search = debounce((query) => { console.log('検索:', query); }, 500);
